C语言单链表的定义和使用

2022-07-25
| C
| 浏览
| 903 字
| 2 分钟

链表是一种常见而又重要的数据结构，本文将通过一个示例介绍C语言中单链表的定义和使用。

单链表结构体声明

#include <stdio.h>
#include <stdlib.h>

struct student
{
    char sno[8];
    char name[20];
    char sex;
    int age;
    double score;
    struct student *next;
};

单链表操作方法声明

//将p所指节点作为第一个节点插入到以head为头指针的单链表中
void insert_node(struct student *head, struct student *p);

//输出以head为头指针的单链表中所有学生的信息
void traverse(struct student *head);

//删除以head为头指针的单链表的第一个节点
void delete_node(struct student *head);

void convert(struct student *head);

主函数

int main(void)
{
    struct student *h, *q;
    int i, n;

    printf("Input the number of students: ");
    scanf("%d", &n);
    getchar(); //读取并丢弃“换行符”

    h = (struct student *)malloc(sizeof(struct student)); //创建头节点
    h->next = NULL;

    printf("Input %d students' information(sex,sno,name,age,score):\n", n);
    for (i = 0; i < n; i++)
    {
        q = (struct student *)malloc(sizeof(struct student)); //创建新节点
        scanf("%c%s%s%d%lf", &q->sex, &q->sno, &q->name, &q->age, &q->score);
        getchar();
        insert_node(h, q);
    }

    traverse(h);
    convert(h);
    traverse(h);
    while (h->next)
    {
        delete_node(h);
    }

    free(h);
    h = NULL;
    return 0;
}

插入节点方法实现

void insert_node(struct student *head, struct student *p)
{
    p->next = head->next;
    head->next = p;
}

遍历链表节点

void traverse(struct student *head)
{
    struct student *p;
    puts("  sno                   name  sex  age   score");
    p = head->next; //p指向单链表的第一个节点
    while (p)
    {
        printf("%s%20s%5c%5d%8.1f\n", p->sno, p->name, p->sex, p->age, p->score);
        p = p->next;
    }
}

删除节点

void delete_node(struct student *head)
{
    struct student *p;
    p = head->next;
    head->next = p->next;
    printf("Deleting the student %s ...\n", p->sno);
    free(p);
    p = NULL;
}

反转单链表（不包括头节点）

// 第一种方法
// 断开链表头，以头插法将原链表数据添加到断开的链表头后
void convert(struct student *head)
{
    struct student *p = head->next;
    head->next = NULL;
    struct student *temp = NULL;
    while (p)
    {
        temp = p->next;
        p->next = head->next;
        head->next = p;
        p = temp;
    }
}

// 第二种方法
// 除链表第一个节点外，后面节点依次以头插法插入到头节点后第一个节点之间
void convert(struct student *head)
{
    struct student *p = head->next;
    struct student *q = NULL;
    while (q = p->next)
    {
        p->next = q->next;
        q->next = head->next;
        head->next = q;
    }
}

综合实例

已知head指向一个带头结点的单向链表，链表中每个结点包含数据域和指针域。请编写程序实现如图所示链表的逆置

原链表：head -> a -> b -> c

逆置后：head -> c -> b -> a

#include <stdio.h>
#include <stdlib.h>

struct node
{
    char data;
    struct node *next;
};

struct node *createList(int num);
void printList(struct node *head);
struct node *reverseList(struct node *head);

int main(void)
{
    struct node *head = createList(3);
    printList(head);
    head = reverseList(head);
    printList(head);
    return 0;
}

struct node *createList(int num) //创建带头结点的链表
{
    struct node *head = (struct node *)malloc(sizeof(struct node));
    head->data = 0;
    head->next = NULL;
    struct node *p = head;
    while (num) //创建头节点之后的节点
    {
        struct node *node = (struct node *)malloc(sizeof(struct node));
        printf("请输入节点的值：");
        scanf("%c", &node->data);
        getchar(); //丢弃换行符
        p->next = node;
        p = node;
        num--;
    }
    p->next = NULL;
    return head;
}

struct node *reverseList(struct node *head)
{
    if (head == NULL || head->next == NULL)
    {
        return head;
    }
    struct node *p = head->next, *nHead = NULL; //头节点不动，逆置后面节点
    while (p != NULL)
    {
        struct node *temp = p;
        p = p->next;
        temp->next = nHead;
        nHead = temp;
    }
    head->next = nHead; //将头节点next指向逆置后的后面节点的头节点，即c节点
    return head;
}

void printList(struct node *head)
{
    struct node *p = head->next; //头节点不打印
    if (!p)
    {
        return;
    }
    while (p)
    {
        printf("%c  ", p->data);
        p = p->next;
    }
    printf("\n");
}
